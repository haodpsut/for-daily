% ---------- 1. VISION ----------
\section*{1. Vision \& Positioning}
\textbf{Chúng tôi xây hệ thống AI Agent tự vận hành, có kiểm soát, cho các workflow tri thức} — bắt đầu từ giáo dục, mở rộng sang du lịch và dịch vụ. Khách hàng trả tiền cho \emph{kết quả vận hành}, không trả tiền cho kiến trúc.
\begin{itemize}
\item \textbf{Thị trường nhìn vào:} AI Agent Systems for Knowledge-Intensive Workflows.
\item \textbf{Tài sản nội bộ:} một framework chuẩn để dựng nhanh AI workflow / agent cho nhiều ngành.
\item \textbf{Không bán} khái niệm ``framework''. \textbf{Không nhảy} multi-agent khi workflow đơn giản đủ dùng.
\end{itemize}

% ---------- 4. 12-MONTH PLAN ----------
\section*{4. 12-month Plan}
\begin{itemize}
\item \textbf{Q1 (M1--M3) — Chọn hẹp.} 1 vertical = \emph{Giáo dục}. 2 use case = \emph{Admissions Agent} + \emph{Internal Knowledge Agent}. 10 discovery call. 1 prototype chạy trên trường thật.
\item \textbf{Q2 (M4--M6) — Design partner.} 2--3 pilot trả phí (tối thiểu 20--30\% giá thật). Hoàn thiện 6 khối framework nội bộ. Đo ROI thật trên 1 trường.
\item \textbf{Q3 (M7--M9) — Packaging.} Chuẩn hóa connector / dashboard / evals / templates cho ngành giáo dục. Mở rộng 5--7 khách. Viết 1--2 case study.
\item \textbf{Q4 (M10--M12) — Vertical 2.} Thử nghiệm \emph{Service / Visitor Agent} cho du lịch trên nền framework sẵn có. Chưa đụng subscription / platform play.
\end{itemize}

% ---------- 5. RULES ----------
\section*{5. Rules of Engagement}
\begin{itemize}
\item \textbf{3 chữ ký / use case:} Hoàng (có khách trả tiền) $+$ Phú (làm được trong X tuần) $+$ Hảo (khớp roadmap). Thiếu 1 $\Rightarrow$ không bán, không build.
\item \textbf{Không đẻ framework trước khi có khách.} Framework chỉ lớn lên từ use case thật, không xây trong chân không.
\item \textbf{Không ôm 2 vertical cùng cường độ} trong năm đầu. Giáo dục ổn rồi mới mở du lịch.
\item \textbf{Đơn giản trước, multi-agent sau.} Single agent + tool + workflow tốt $>$ multi-agent khi chưa cần.
\item \textbf{Không làm outsourcing đội lốt platform.} Nếu mỗi khách một kiểu, không tái sử dụng $\Rightarrow$ từ chối hoặc tính giá agency.
\end{itemize}
